typedef enum {
    PKFS_IDLE = 0,
    PKFS_SUCKER_ON,           // Step1: 开吸盘
    PKFS_ARM_POSE,            // Step2: 摆取KFS姿态 (复用 arm.c mode 0)
    PKFS_ARM_PUSH,            // Step3: 机械臂前推 s递增 (复用 arm.c 逆解)
    PKFS_ARM_RETRACT,         // Step4: 机械臂三阶段回收 (复用 arm.c mode 3)
    PKFS_DONE,
    PKFS_TIMEOUT,
} PickKfsState;

typedef struct {
    /* ==== 触发 (Keil watch 写 0/1) ==== */
    uint8_t  trigger;          // 写1→启动自动化 (代码自动清零)
    uint8_t  step_mode;        // 0=全自动连续  1=单步逐帧
    uint8_t  step_next;        // 写1→执行1帧→代码自动清零
    uint8_t  abort;            // 写1→紧急停止→回 IDLE

    /* ==== 只读状态 ==== */
    PickKfsState state;        // 当前状态
    uint16_t frame_cnt;        // 当前状态已执行帧数 (×5ms)
    uint8_t  retract_phase;    // 回收阶段: 0=A 1=B 2=C

    /* ==== 高度选择 0=低(0.23) 1=中(0.435) 2=高(0.65) ==== */
    uint8_t  height_sel;

    /* ==== Step1: 吸盘延时 (ms) ==== */
    uint16_t sucker_delay_ms;  // 默认200

    /* ==== Step3: 前推 s 范围 (默认 0.32 → 0.60, 步长0.005) ==== */
    float    push_s[3];         // [0]=起始s, [1]=目标s, [2]=每帧增量

    /* ==== Step4: 回收完成条件 ==== */
    float    retract_done_pos; // arm_2 < 此值完成, 默认-4.2
} PickKfsDebug;

2. 单步逐帧调试设计

STEP-DEBUG 核心机制: step_next=1 → 执行1帧 → 代码自动清零

2.1 两种运行模式

模式step_mode行为适用场景
全自动 0 trigger=1 启动后，四个步骤自动连续执行到底，trigger 完成后自动归零 正常比赛/测试运行
单步逐帧 1 trigger=1 进入 Step1 后停止，每次在 Keil 写 step_next=1 才执行1帧(5ms)，执行完后 step_next 自动归零。到位后自动切换状态并停住等待下一次 step_next 调参、排查关节运动、验证到位阈值

2.2 单步调试操作序列

// Keil Debug → Watch 窗口添加 g_dbg

// === 步骤1: 进入调试模式 ===
g_dbg.step_mode = 1;       // 使能单步模式 (只需设置一次)
g_dbg.trigger = 1;         // 触发自动化 → 进入 PKFS_SUCKER_ON, 停住

// === 步骤2: 逐帧推进吸盘阶段 ===
g_dbg.step_next = 1;      // 执行1帧 → step_next 自动归零
g_dbg.step_next = 1;      // 再执行1帧
// ... 反复写 g_dbg.step_next=1 直到 frame_cnt 达到 sucker_delay_ms/5
// 到位后自动进入 PKFS_ARM_POSE, frame_cnt 归零, 停住

// === 步骤3: 逐帧推进摆姿态阶段 ===
g_dbg.step_next = 1;      // 执行1帧逆解+发送 → 电机微动
g_dbg.step_next = 1;      // 再1帧, 观察 kfs_arm_1/2/3.position
// ... 反复写直到3关节到位 → 自动进入 PKFS_ARM_PUSH

// === 步骤4: 逐帧推进前推阶段 ===
g_dbg.step_next = 1;      // s+0.005 → 逆解 → 发送, 观察 s 变化
// ... 反复写 g_dbg.step_next=1 直到 push_s >= push_s[1] → 进入 PKFS_ARM_RETRACT

// === 步骤5: 逐帧推进回收阶段 ===
g_dbg.step_next = 1;      // 观察 retract_phase 变化: 0→1→2
// ... 反复写直到 arm_2.pos < retract_done_pos → PKFS_DONE

// === 随时终止 ===
g_dbg.abort = 1;          // 任何时候写1 → 立刻回 IDLE

2.3 单步消费逻辑 (consume_step 函数)

/* 
 * 返回值: 1=本帧可以执行, 0=等待用户写 step_next
 * 仅在 step_mode==1 时调用
 */
static uint8_t consume_step(void) {
    if (g_dbg.step_next) {
        g_dbg.step_next = 0;   // 消费信号, 防止误触发下一帧
        return 1;
    }
    return 0;
}

#include "kfs.h"           // kfs_arm_1, kfs_arm_2, kfs_arm_3

/* ===== 高度配置表 (arm.c mode 0/5/6) ===== */
static const float HEIGHT_CFG[3] = {
    0.230f, // LOW  → arm.c L119
    0.435f, // MID  → arm.c L237
    0.650f, // HIGH → arm.c L251
};

/* ===== 全局调试结构体 (volatile → Keil Watch 可实时读写) ===== */
static volatile PickKfsDebug g_dbg;

/* ===== 运行时变量 ===== */
static float g_push_s;         // 前推过程 s 值
static float g_h_target;       // 目标高度
static uint32_t g_t0;         // 当前状态启动时刻(ms)
static float g_kfs_1[5];      // in_place 中间变量
static float g_kfs_2[5];

/* ===== 辅助 ===== */
static uint8_t consume_step(void) {
    if (g_dbg.step_next) { g_dbg.step_next = 0; return 1; }
    return 0;
}

static void sucker_set(uint8_t on) {
    GPIO_PinState s = on ? GPIO_PIN_SET : GPIO_PIN_RESET;
    HAL_GPIO_WritePin(GPIOE, GPIO_PIN_14, s);
    HAL_GPIO_WritePin(GPIOC, GPIO_PIN_12, s);
}

/* ===== 单帧: 逆解→补偿→三关节发送 (复用 arm.c L120-125) ===== */
static void arm_send_pose(float s, float h) {
    arm.mode = 0;                               // arm.c L117
    arm.s    = s;
    arm.h    = h;
    Arm_Inverse_Solution(&arm);              // arm.c L120
    Torque_Comp_global(&kfs_arm_1, &kfs_arm_2, &kfs_arm_3); // arm.c L121

    // 关节1 → arm.c L123 (KP/KD 硬编码 = 70.0/5.0)
    kfs_arm_1.set_mit_data(&kfs_arm_1,
        angles.theta_1 + 0.43f, 0.0f, PUSH_KP_ARM1, PUSH_KD_ARM1,
        -0.25f * Torque.Torque_1);

    // 关节2 → arm.c L124
    kfs_arm_2.set_mit_data(&kfs_arm_2,
        -angles.theta_2, 0.0f, PUSH_KP_ARM2, PUSH_KD_ARM2,
        -0.6f * Torque.Torque_2);

    // 关节3 → arm.c L125
    kfs_arm_3.set_mit_data(&kfs_arm_3,
        25.0f * (PI/2.0f + kfs_arm_1.position - 0.43f
                 + kfs_arm_2.position - PI * 0.08f),
        0.0f, 0.2f, 0.15f, 0.8f * Torque.Torque_3);
}

/* ===== Step4 回收单帧 (完整复制 arm.c L142-168 mode 3) ===== */
static void arm_retract_tick(void) {
    Torque_Comp_global(&kfs_arm_1, &kfs_arm_2, &kfs_arm_3);

    if (kfs_arm_2.position > -1.6f) {                          // A段 arm.c L146
        g_dbg.retract_phase = 0;
        in_place(-1.6f, &g_kfs_2[0], kfs_arm_2.position, &g_kfs_2[1],
                  0.6f, &g_kfs_2[2], 30.0f, &g_kfs_2[3], 5.0f, 4.0f,
                  &g_kfs_2[4], -0.5f*Torque.Torque_2, -2.0f*Torque.Torque_2, 0.0f);
        kfs_arm_2.set_mit_data(&kfs_arm_2,
            g_kfs_2[0], g_kfs_2[1], g_kfs_2[2], g_kfs_2[3], g_kfs_2[4]);
    }
    else if (kfs_arm_2.position > -2.92f) {                   // B段 arm.c L152
        g_dbg.retract_phase = 1;
        in_place(0.2f, &g_kfs_1[0], kfs_arm_1.position, &g_kfs_1[1],
                  2.0f, &g_kfs_1[2], 55.0f, &g_kfs_1[3], 5.0f, 5.0f,
                  &g_kfs_1[4], -0.5f*Torque.Torque_1, -0.5f*Torque.Torque_1, 0.3f);
        kfs_arm_1.set_mit_data(&kfs_arm_1,
            g_kfs_1[0], g_kfs_1[1], g_kfs_1[2], g_kfs_1[3], g_kfs_1[4]);
        in_place(-2.92f, &g_kfs_2[0], kfs_arm_2.position, &g_kfs_2[1],
                  0.1f, &g_kfs_2[2], 30.0f, &g_kfs_2[3], 5.0f, 4.0f,
                  &g_kfs_2[4], -0.2f*Torque.Torque_2, -1.8f*Torque.Torque_2, 0.0f);
        kfs_arm_2.set_mit_data(&kfs_arm_2,
            g_kfs_2[0], g_kfs_2[1], g_kfs_2[2], g_kfs_2[3], g_kfs_2[4]);
    }
    else {                                                      // C段 arm.c L161
        g_dbg.retract_phase = 2;
        kfs_arm_3.set_mit_data(&kfs_arm_3,
            -52.0f, 0.0f, 0.13f, 0.6f, 1.0f*Torque.Torque_3);
        kfs_arm_1.set_mit_data(&kfs_arm_1,
            0.6f, 0.0f, 60.0f, 5.0f, -0.8f*Torque.Torque_1);
        in_place_1(-4.3f, &g_kfs_2[0], kfs_arm_2.position, &g_kfs_2[1],
                    -0.0f, &g_kfs_2[2], 65.0f, &g_kfs_2[3], 3.0f, 5.0f,
                    &g_kfs_2[4], -0.12f*Torque.Torque_2, -1.0f*Torque.Torque_2, 0.28f);
        kfs_arm_2.set_mit_data(&kfs_arm_2,
            g_kfs_2[0], g_kfs_2[1], g_kfs_2[2], g_kfs_2[3], g_kfs_2[4]);
    }
}

PickKfsState PickKfsAuto_State(void) {
    return (PickKfsState)g_dbg.state;
}

uint8_t PickKfsAuto_IsIdle(void) {
    return (g_dbg.state == PKFS_IDLE || g_dbg.state == PKFS_DONE);
}

void PickKfsAuto_Run(void) {
    /* -- abort 优先 -- */
    if (g_dbg.abort) {
        sucker_set(0);
        g_dbg.state = PKFS_IDLE;
        g_dbg.abort = 0;
        g_dbg.trigger = 0;
        return;
    }

    /* -- 全自动模式无 trigger 则跳过 -- */
    if (!g_dbg.step_mode && !g_dbg.trigger) return;

    uint32_t now = common_now_ms();

    switch (g_dbg.state) {

    case PKFS_IDLE:
        if (g_dbg.trigger) {
            g_dbg.trigger = 0;   // 消费 trigger
            g_h_target  = HEIGHT_CFG[g_dbg.height_sel];
            g_push_s    = g_dbg.push_s[0];
            g_dbg.frame_cnt = 0;
            g_t0        = now;
            g_dbg.state = PKFS_SUCKER_ON;
            if (g_dbg.step_mode) return;  // 单步: 停在这里, 用户看到状态切换
        }
        break;

    case PKFS_SUCKER_ON:
        if (g_dbg.step_mode && !consume_step()) return;
        sucker_set(1);
        g_dbg.frame_cnt++;
        if ((g_dbg.frame_cnt * 5) >= g_dbg.sucker_delay_ms ||
            now - g_t0 > TIMEOUT[0]) {
            g_dbg.state = PKFS_ARM_POSE;
            g_dbg.frame_cnt = 0;  g_t0 = now;
            if (g_dbg.step_mode) return;
        }
        break;

    case PKFS_ARM_POSE:
        if (g_dbg.step_mode && !consume_step()) return;
        arm_send_pose(g_dbg.push_s[0], g_h_target);
        g_dbg.frame_cnt++;
        if ((ABS(kfs_arm_1.position - (angles.theta_1 + 0.43f)) < POSE_TOL[0]) &&
            (ABS(kfs_arm_2.position + angles.theta_2) < POSE_TOL[1]) &&
            (ABS(kfs_arm_3.position - 25.0f*(PI/2.0f + kfs_arm_1.position
                - 0.43f + kfs_arm_2.position - PI * 0.08f)) < POSE_TOL[2])) {
            g_dbg.state = PKFS_ARM_PUSH;
            g_dbg.frame_cnt = 0;  g_t0 = now;
            if (g_dbg.step_mode) return;
        }
        if (now - g_t0 > TIMEOUT[1]) {
            g_dbg.state = PKFS_TIMEOUT;
        }
        break;

    case PKFS_ARM_PUSH:
        if (g_dbg.step_mode && !consume_step()) return;
        g_push_s = clampf(g_push_s + g_dbg.push_s[2],
                          g_dbg.push_s[0], g_dbg.push_s[1]);
        arm_send_pose(g_push_s, g_h_target);
        g_dbg.frame_cnt++;
        if (g_push_s >= g_dbg.push_s[1] ||
            now - g_t0 > TIMEOUT[2]) {
            g_dbg.state = PKFS_ARM_RETRACT;
            g_dbg.frame_cnt = 0;  g_t0 = now;
            if (g_dbg.step_mode) return;
        }
        break;

    case PKFS_ARM_RETRACT:
        if (g_dbg.step_mode && !consume_step()) return;
        arm_retract_tick();
        g_dbg.frame_cnt++;
        if (kfs_arm_2.position < g_dbg.retract_done_pos ||
            now - g_t0 > TIMEOUT[3]) {
            g_dbg.state = PKFS_DONE;
            sucker_set(0);
            if (g_dbg.step_mode) return;
        }
        break;

    case PKFS_DONE:
        g_dbg.state = PKFS_IDLE;
        break;

    case PKFS_TIMEOUT:
        g_dbg.state = PKFS_IDLE;
        break;

    default: break;
    }
}

6.1 Arm_task 入口集成 (arm.c L95-97)

Arm_task 是所有机械臂控制的唯一入口（HF_Task 3个zone分支 + auto_mode 均调用它）。在入口处检查自动化是否运行，一处修改覆盖所有调用方。

6.2 为什么放在 Arm_task 入口

Arm_task 在 HF_Task() 中被 3 个 zone 分支调用，也在 channel==1(自动模式) 中被调用。在入口处做互斥，一处修改覆盖所有调用方，不需在 Can_Task.c 重复判断。

6.3 调用链

HF_Task (200Hz)
  ├─ zone==0: Arm_task() → IsIdle()? → Run() 或 原有逻辑
  ├─ zone==1: Arm_task() → IsIdle()? → Run() 或 原有逻辑
  ├─ zone==2: Arm_task() → IsIdle()? → Run() 或 原有逻辑
  └─ channel==1: Arm_task() → IsIdle()? → Run() 或 auto_mission()等

所有路径自动拦截, Can_Task.c 零修改
