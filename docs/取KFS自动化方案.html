/**
 * @file  pick_kfs_auto.h
 * @brief 取KFS全流程自动化
 *
 * 设计原则:
 *   1. arm.c 零修改 — 所有姿态参数直接复制 arm.c L115-258
 *   2. motor_control.c 零修改 — in_place/in_place_1 直接调用
 *   3. 参考 auto_mode.c auto_flag 步进模式
 *   4. 参考 arm.c mode 3 多阶段回收模式
 */
#ifndef __PICK_KFS_AUTO_H__
#define __PICK_KFS_AUTO_H__

#include "arm.h"
#include "common.h"

/* KFS货架高度 (对应 arm mode 0/5/6 的 h 值) */
typedef enum {
    PKFS_H_LOW  = 0,   // h=0.230 → arm mode 0
    PKFS_H_MID  = 1,   // h=0.435 → arm mode 5
    PKFS_H_HIGH = 2,   // h=0.650 → arm mode 6
} PickKfsHeight;

typedef enum {
    PKFS_IDLE = 0,
    PKFS_SUCKER_ON,    // Step1: 复用 kfs.c L27-28 GPIO
    PKFS_ARM_POSE,     // Step2: 复用 arm.c L117-125 mode 0
    PKFS_ARM_PUSH,     // Step3: 复用 arm.c L28-36 逆解 + s递增
    PKFS_ARM_RETRACT,  // Step4: 复用 arm.c L142-168 mode 3
    PKFS_DONE,
    PKFS_ERROR,
} PickKfsState;

void PickKfsAuto_Init(void);
void PickKfsAuto_Run(void);
void PickKfsAuto_Start(PickKfsHeight h);
void PickKfsAuto_Abort(void);
PickKfsState PickKfsAuto_GetState(void);
uint8_t      PickKfsAuto_IsBusy(void);

#endif

#include "pick_kfs_auto.h"
#include "kfs.h"           // kfs_arm_1/2/3, kfs_catch 全局变量
#include "motor_control.h" // in_place(), in_place_1()

/* ===== 参数表: 来自 arm.c mode 0/5/6 ===== */
static const float HEIGHT_CFG[3] = {
    0.230f, // LOW  → arm.c L119
    0.435f, // MID  → arm.c L237
    0.650f, // HIGH → arm.c L251
};

static const float S_INIT = 0.32f;  // arm.c L118/236/250
static const float S_MAX  = 0.60f;  // arm.c L118/236/250
static const float PUSH_STEP = 0.005f;

/* ===== 上下文 ===== */
static struct {
    PickKfsState state;
    float  h_target;
    float  push_s;
    uint32_t t0;
    uint8_t  retract_phase;  // 0=A, 1=B, 2=C (参考 arm mode 3)
    uint8_t  error_code;
} g;

/* 高度选择 */
static float cfg_h(PickKfsHeight h) {
    return HEIGHT_CFG[h];
}

/* ===== 原子操作: 开吸盘 (复用 kfs.c L27-28) ===== */
static void sucker_set(uint8_t on) {
    GPIO_PinState s = on ? GPIO_PIN_SET : GPIO_PIN_RESET;
    HAL_GPIO_WritePin(GPIOE, GPIO_PIN_14, s);
    HAL_GPIO_WritePin(GPIOC, GPIO_PIN_12, s);
}

/* ===== 机械臂三关节发送 (复制 arm.c L120-125) ===== */
static void arm_send_kfs_pose(float s, float h) {
    extern Arm_Info_TypeDef arm;
    extern ArmAnglesTypeDef angles;
    extern ArmTorqueTypeDef Torque;

    arm.mode = 0;           // 逆解 case 0 → arm.c L28-36
    arm.s    = s;
    arm.h    = h;

    angles = Arm_Inverse_Solution(&arm);           // arm.c L13
    Torque = Torque_Comp_global(&kfs_arm_1,       // arm.c L70
                                  &kfs_arm_2,
                                  &kfs_arm_3);

    // arm_1 → arm.c L123
    kfs_arm_1.set_mit_data(&kfs_arm_1,
        angles.theta_1 + 0.43f, 0.0f,
        70.0f, 5.0f, -0.25f * Torque.Torque_1);
    // arm_2 → arm.c L124
    kfs_arm_2.set_mit_data(&kfs_arm_2,
        -angles.theta_2, 0.0f,
        50.0f, 4.5f, -0.6f * Torque.Torque_2);
    // arm_3 → arm.c L125
    kfs_arm_3.set_mit_data(&kfs_arm_3,
        25.0f * (PI/2.0f + kfs_arm_1.position
            - 0.43f + kfs_arm_2.position - PI * 0.08f),
        0.0f, 0.2f, 0.15f, 0.8f * Torque.Torque_3);
}

/* ===== S2 到位判断 (参考 auto_mode.c L46-49) ===== */
static uint8_t arm_pose_done(void) {
    return (ABS(kfs_arm_1.position - g.arm1_target) < 0.03f) &&
           (ABS(kfs_arm_2.position - g.arm2_target) < 0.03f) &&
           (ABS(kfs_arm_3.position - g.arm3_target) < 2.0f);
}

/* ===== S4 回收 (直接复制 arm.c L142-168 mode 3) ===== */
static void arm_retract_tick(void) {
    extern ArmTorqueTypeDef Torque;
    float kfs_1[5], kfs_2[5];
    Torque = Torque_Comp_global(&kfs_arm_1, &kfs_arm_2, &kfs_arm_3);

    if (kfs_arm_2.position > -1.6f) {                    // arm.c L146
        in_place(-1.6f, &kfs_2[0], kfs_arm_2.position, &kfs_2[1],
                  0.6f, &kfs_2[2], 30.0f, &kfs_2[3], 5.0f, 4.0f,
                  &kfs_2[4], -0.5f*Torque.Torque_2, -2.0f*Torque.Torque_2, 0.0f);
        kfs_arm_2.set_mit_data(&kfs_arm_2,
            kfs_2[0], kfs_2[1], kfs_2[2], kfs_2[3], kfs_2[4]);
    }
    else if (kfs_arm_2.position > -2.92f) {             // arm.c L152
        in_place(0.2f, &kfs_1[0], kfs_arm_1.position, &kfs_1[1],
                  2.0f, &kfs_1[2], 55.0f, &kfs_1[3], 5.0f, 5.0f,
                  &kfs_1[4], -0.5f*Torque.Torque_1, -0.5f*Torque.Torque_1, 0.3f);
        kfs_arm_1.set_mit_data(&kfs_arm_1,
            kfs_1[0], kfs_1[1], kfs_1[2], kfs_1[3], kfs_1[4]);
        in_place(-2.92f, &kfs_2[0], kfs_arm_2.position, &kfs_2[1],
                  0.1f, &kfs_2[2], 30.0f, &kfs_2[3], 5.0f, 4.0f,
                  &kfs_2[4], -0.2f*Torque.Torque_2, -1.8f*Torque.Torque_2, 0.0f);
        kfs_arm_2.set_mit_data(&kfs_arm_2,
            kfs_2[0], kfs_2[1], kfs_2[2], kfs_2[3], kfs_2[4]);
    }
    else {                                            // arm.c L161
        kfs_arm_3.set_mit_data(&kfs_arm_3,
            -52.0f, 0.0f, 0.13f, 0.6f, 1.0f*Torque.Torque_3);
        kfs_arm_1.set_mit_data(&kfs_arm_1,
            0.6f, 0.0f, 60.0f, 5.0f, -0.8f*Torque.Torque_1);
        in_place_1(-4.3f, &kfs_2[0], kfs_arm_2.position, &kfs_2[1],
                    -0.0f, &kfs_2[2], 65.0f, &kfs_2[3], 3.0f, 5.0f,
                    &kfs_2[4], -0.12f*Torque.Torque_2, -1.0f*Torque.Torque_2, 0.28f);
        kfs_arm_2.set_mit_data(&kfs_arm_2,
            kfs_2[0], kfs_2[1], kfs_2[2], kfs_2[3], kfs_2[4]);
    }
}

/* ===== 主状态机 ===== */
void PickKfsAuto_Run(void) {
    uint32_t now = common_now_ms();
    switch (g.state) {
    case PKFS_SUCKER_ON:
        sucker_set(1);
        if (now - g.t0 >= 200) { g.state = PKFS_ARM_POSE; g.t0 = now; }
        break;
    case PKFS_ARM_POSE:
        arm_send_kfs_pose(S_INIT, g.h_target);
        if (arm_pose_done()) {
            g.push_s = S_INIT; g.state = PKFS_ARM_PUSH; g.t0 = now;
        }
        break;
    case PKFS_ARM_PUSH:
        g.push_s = clampf(g.push_s + PUSH_STEP, S_INIT, S_MAX);
        arm_send_kfs_pose(g.push_s, g.h_target);
        if (g.push_s >= S_MAX) { g.state = PKFS_ARM_RETRACT; g.t0 = now; }
        break;
    case PKFS_ARM_RETRACT:
        arm_retract_tick();
        if (kfs_arm_2.position < -4.2f) { g.state = PKFS_DONE; }
        break;
    }
}
